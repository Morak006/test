#!/bin/bash

# СТРОКА 2: Простая проверка прав (совместима со всеми оболочками)
if [ "$(id -u)" != "0" ]; then
   echo "Запустите от root (sudo)"
   exit 1
fi

# Ввод данных
read -p "IP eth1 [172.16.4.1/28]: " A1
A1=${A1:-172.16.4.1/28}
read -p "IP eth2 [172.16.5.1/28]: " A2
A2=${A2:-172.16.5.1/28}

# Расчет сетей
N1=$(echo $A1 | cut -d'.' -f1-3).0/$(echo $A1 | cut -d'/' -f2)
N2=$(echo $A2 | cut -d'.' -f1-3).0/$(echo $A2 | cut -d'/' -f2)

# Настройка Hostname и Сети
hostnamectl set-hostname isp
echo -e "auto lo\niface lo inet loopback\n\nauto eth0\niface eth0 inet dhcp\n\nauto eth1\niface eth1 inet static\n    address $A1\n\nauto eth2\niface eth2 inet static\n    address $A2" > /etc/network/interfaces

# Forwarding
echo 1 > /proc/sys/net/ipv4/ip_forward
sed -i 's/#net.ipv4.ip_forward=1/net.ipv4.ip_forward=1/' /etc/sysctl.conf 2>/dev/null

# Рестарт сети
systemctl restart networking

# NAT (iptables)
iptables -t nat -F
iptables -t nat -A POSTROUTING -s $N1 -o eth0 -j MASQUERADE
iptables -t nat -A POSTROUTING -s $N2 -o eth0 -j MASQUERADE

# Сохранение правил
mkdir -p /root
iptables-save > /root/rules

# СТРОКИ 63-64 (Cron): Новый максимально надежный метод
# Мы просто создаем файл в cron.d, это надежнее, чем править crontab -e
echo "@reboot root /sbin/iptables-restore < /root/rules" > /etc/cron.d/iptables_restore
chmod 644 /etc/cron.d/iptables_restore

# Время
apt-get update && apt-get install -y tzdata
timedatectl set-timezone Europe/Moscow

echo "Done!"
