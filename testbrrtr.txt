#!/bin/bash

echo "=== Генератор конфигурации BR-RTR (Branch) ==="

# 1. Сбор данных интерфейсов
read -p "Внешний IP BR (isp interface) [172.16.5.14/28]: " BR_ISP_ADDR
BR_ISP_ADDR=${BR_ISP_ADDR:-172.16.5.14/28}

read -p "Шлюз провайдера для BR [172.16.5.1]: " BR_ISP_GW
BR_ISP_GW=${BR_ISP_GW:-172.16.5.1}

read -p "IP интерфейса BR-NET [192.168.200.1/27]: " BR_NET_ADDR
BR_NET_ADDR=${BR_NET_ADDR:-192.168.200.1/27}

# 2. Данные туннеля
read -p "Внутренний IP туннеля (Tunnel 0) [10.10.10.2/30]: " BR_TUN_ADDR
BR_TUN_ADDR=${BR_TUN_ADDR:-10.10.10.2/30}

read -p "Удаленный внешний IP для GRE (HQ IP) [172.16.4.14]: " HQ_REMOTE
HQ_REMOTE=${HQ_REMOTE:-172.16.4.14}

# 3. Настройка OSPF сетей
echo -e "\n--- Настройка сетей OSPF ---"
read -p "Сеть туннеля [10.10.10.0/30]: " BR_OSPF_NET_TUN
BR_OSPF_NET_TUN=${BR_OSPF_NET_TUN:-10.10.10.0/30}

read -p "Сеть BR-NET для анонса [192.168.200.0/27]: " BR_OSPF_NET_LOCAL
BR_OSPF_NET_LOCAL=${BR_OSPF_NET_LOCAL:-192.168.200.0/27}

# Подготовка вспомогательных переменных
BR_ISP_IP_ONLY=$(echo $BR_ISP_ADDR | cut -d'/' -f1)
BR_TUN_IP_ONLY=$(echo $BR_TUN_ADDR | cut -d'/' -f1)

echo -e "\n--- СКОПИРУЙТЕ КОМАНДЫ НИЖЕ ---\n"

cat <<EOF
conf t
hostname br-rtr
ip domain-name au-team.irpo

int isp 
 description "connect ISP"
 ip address $BR_ISP_ADDR
 ip nat outside
ex

port te0
 service-instance te0/isp
  encapsulation untagged
  connect ip interface isp
 ex
ex

ip route 0.0.0.0/0 $BR_ISP_GW

interface BR-NET
 description "Connect BR-NET"
 ip address $BR_NET_ADDR
 ip nat inside
exit

port te1
 service-instance te1/BR-NET
  encapsulation untagged
  connect ip interface BR-NET
 exit
exit

username net_admin password P@\$\$word role admin

interface tunnel.0
 description "GRE-to-HQ"
 ip address $BR_TUN_ADDR
 ip tunnel $BR_ISP_IP_ONLY $HQ_REMOTE mode gre
 ip ospf authentication message-digest
 ip ospf message-digest-key 1 md5 P@\$\$word
exit

router ospf 1
 ospf router-id $BR_TUN_IP_ONLY
 passive-interface default
 no passive-interface tunnel.0
 network $BR_OSPF_NET_TUN area 0
 network $BR_OSPF_NET_LOCAL area 0
exit

ip nat pool BR_POOL 192.168.200.1-192.168.200.254
ip nat source dynamic inside-to-outside pool BR_POOL overload interface isp

do write memory
EOF

echo -e "\n--- КОНЕЦ КОНФИГУРАЦИИ ---"
