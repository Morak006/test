! --- Вход в режим конфигурации ---
configure

! --- Настройка системных параметров ---
set system hostname hq-rtr
set system timezone Europe/Moscow

! --- Настройка интерфейсов ---
! eth0 — смотрит на ISP
set interfaces ethernet eth0 address 172.16.4.2/28
set interfaces ethernet eth0 description "Link to ISP"

! eth1 — внутренняя сеть HQ
set interfaces ethernet eth1 address 192.168.100.1/26
set interfaces ethernet eth1 description "HQ Local Network"

! --- Статический маршрут по умолчанию (шлюз — ISP) ---
set protocols static route 0.0.0.0/0 next-hop 172.16.4.1

! --- Настройка NAT (чтобы локалка ходила в инет) ---
set nat source rule 10 outbound-interface eth0
set nat source rule 10 source address 192.168.100.0/26
set nat source rule 10 translation address masquerade

! --- Настройка SSH для удаленного управления ---
set service ssh port 22
set service ssh listen-address 192.168.100.1

! --- Применение и сохранение ---
commit
save
exit
